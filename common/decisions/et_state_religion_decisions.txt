# Hussite
et_adopt_hussite_state_religion_decision = {
    is_shown = {
        religion_percent_country = {
            target = rel:hussite
            value >= 0.51
        }
        NOT = { country_has_state_religion = rel:hussite }
    }
    possible = {
        religion_percent_country = {
            target = rel:hussite
            value >= 0.51
        }
        NOT = { country_has_state_religion = rel:hussite }
        has_variable = newly_formed_country
        NOT = { has_variable = newly_formed_country_cooldown }
    }
    when_taken = {
        set_state_religion = rel:hussite
    }
    ai_chance = { value = 100 }
}

### Victoria3Timeline: Majority Religion Adoption Decisions
### Each decision appears when a single religion reaches >= 51% of population
### and the country does not already have that religion as its state religion.
### AI will always take it once possible.

et_adopt_sunni_state_religion_decision = {
    is_shown = {
        religion_percent_country = {
            target = rel:sunni
            value >= 0.51
        }
        NOT = { country_has_state_religion = rel:sunni }
    }

    possible = {
        religion_percent_country = {
            target = rel:sunni
            value >= 0.51
        }
        NOT = { country_has_state_religion = rel:sunni }
        has_variable = newly_formed_country
        NOT = { has_variable = newly_formed_country_cooldown }
    }

    when_taken = {
        set_state_religion = rel:sunni
    }

    ai_chance = { value = 100 }
}
