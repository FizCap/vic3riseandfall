on_monthly_pulse = {
    on_actions = {
        et_annex_decentralized_monthly
    }
}

et_annex_decentralized_monthly = {
    trigger = {
        always = yes
    }
    effect = {
        if = {
            limit = {
                year <= 1447
                has_game_rule = annexation_unlimited
            }
            trigger_event = annex_decentralized_unlimited.1
        }
        else_if = {
            limit = {
                year <= 1447
                has_game_rule = annexation_limited
            }
            trigger_event = annex_decentralized_limited.1
        }
    }
}
