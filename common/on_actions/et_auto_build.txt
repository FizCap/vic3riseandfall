on_monthly_pulse_country = {
    on_actions = {
        et_auto_build_monthly_gov
        et_auto_build_monthly_rail
        et_auto_build_monthly_port
        et_auto_build_monthly_power
    }
}


et_auto_build_monthly_gov = {
    trigger = {
        has_variable = et_auto_build_gov_cooldown
    }
    effect = {
        if = {
            limit = {
                NOT = {
                    any_scope_state = {
                        b:building_government_administration ?= {
                            is_under_construction = yes
                        }
                    }
                }
            }
            et_manual_build_government_administration = yes
        }
    }
}
