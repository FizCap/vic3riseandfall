on_monthly_pulse_country = {
    on_actions = {
        et_auto_build_monthly_gov
        et_auto_build_monthly_rail
        et_auto_build_monthly_port
        et_auto_build_monthly_power
        et_auto_build_monthly_construction
        et_dynamic_country_ai_monthly
        et_dynamic_country_ai_monthly_unrecognized
    }
}


et_auto_build_monthly_gov = {
    trigger = {
        has_variable = et_auto_build_gov_cooldown
    }
    effect = {
        if = {
            limit = {
                NOT = {
                    any_scope_state = {
                        b:building_government_administration ?= {
                            is_under_construction = yes
                        }
                    }
                }
            }
            et_manual_build_government_administration = yes
        }
    }
}

et_dynamic_country_ai_monthly = {
    trigger = {
        is_player = no
        has_game_rule = ai_improvement_enabled
        is_at_war = no
        is_subject = no
        is_country_type = recognized
    }
    effect = {
        et_dynamic_country_ai_recognized = yes
    }
}
