on_game_started = {
    on_actions = {
        et_auto_build_startup
    }
}

on_monthly_pulse_country = {
    on_actions = {
        et_auto_build_monthly
    }
}

et_auto_build_startup = {
    effect = {
        trigger_event = { id = et_auto_build.1 }
    }
}

et_auto_build_monthly = {
    trigger = {
        has_modifier = et_auto_build_enabled
    }
    effect = {
        et_auto_build_all = yes
    }
}
