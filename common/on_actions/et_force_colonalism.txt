force_colonialism = {
  trigger = { year = 1445 }
  effect = {
    every_country = {
      limit = { 
        OR = {
          country_has_primary_culture = cu:british
          country_has_primary_culture = cu:spanish
          country_has_primary_culture = cu:portuguese
          country_has_primary_culture = cu:french
          country_has_primary_culture = cu:dutch
        }
      }
      activate_law = law_type:law_slave_trade
      activate_law = law_type:law_colonial_exploitation
      add_modifier = {
        name = ageofcolonialism_modfier
        years = 50
      }
      # Add declared interest in key colonial regions
      add_declared_interest = region_canada
      add_declared_interest = region_new_england
      add_declared_interest = region_dixie
      add_declared_interest = region_mexico
      add_declared_interest = region_pacific_coast
      add_declared_interest = region_central_america
      add_declared_interest = region_caribbean
      add_declared_interest = region_gran_colombia
      add_declared_interest = region_brazil
      add_declared_interest = region_andes
      add_declared_interest = region_la_plata
      add_declared_interest = region_north_africa
      add_declared_interest = region_senegal
      add_declared_interest = region_niger
      add_declared_interest = region_ethiopia
      add_declared_interest = region_congo
      add_declared_interest = region_zanj
      add_declared_interest = region_southern_africa
      add_declared_interest = region_indonesia
      add_declared_interest = region_oceania
    }
    every_country = {
      limit = { is_player = yes }
      trigger_event = { id = et_force_colonialism.1 }
    }
  }
}

force_colonialism_russia = {
  trigger = { year = 1581 }
  effect = {
    every_country = {
      limit = {
        country_has_primary_culture = cu:russian
      }
      activate_law = law_type:law_frontier_colonization
    }
    every_country = {
      limit = { is_player = yes }
      trigger_event = { id = et_force_colonialism.2 }
    }
  }
}
