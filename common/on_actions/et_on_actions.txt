# --- Event Definitions ---

et_austria_burgundy_union = {
    trigger = {
        year = 1477
        exists = c:AUS
        exists = c:X59
        c:X59 = { NOT = { is_player = yes } }
    }
    effect = {
        et_austria_burgundy_union_effect = yes
    }
}

et_castille_aragon_union = {
    trigger = {
        year = 1479
        exists = c:CAS
        exists = c:X22
        c:X22 = { NOT = { is_player = yes } }
    }
    effect = {
        et_castille_aragon_union_effect = yes
    }
}
