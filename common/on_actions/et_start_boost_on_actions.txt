# Economic Boost Game Rule Effects
start_boost = {
  trigger = { has_game_rule = economic_boost_high }
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier
        days = 365
      }
    }
  }
}

start_boost2 = {
  trigger = {
    OR = {
      has_game_rule = economic_boost_medium
      has_game_rule = economic_boost_high
    }
    game_date <= 1850.1.1
  }
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier2
        days = 365
      }
    }
  }
}

start_boost3 = {
  trigger = {
    OR = {
      has_game_rule = economic_boost_medium
      has_game_rule = economic_boost_high
    }
  }
  effect = {
    every_country = {
      every_scope_state = {
        limit = { 
          is_coastal = yes 
          owner = { has_technology_researched = admiralty }
        }
        create_building = {
          building = "building_naval_base"
          level = 1
          reserves = 1
        }
      }
      capital = {
        create_building = {
          building = "building_construction_sector"
          level = 1
          reserves = 1
        }
        create_building = {
          building = "building_barracks"
          level = 1
          reserves = 1
        }
        create_building = {
          building = "building_trade_center"
          level = 1
          reserves = 1
        }
      }
    }
  }
}

start_boost4 = {
  trigger = {
    OR = {
      has_game_rule = economic_boost_medium
      has_game_rule = economic_boost_high
    }
  }
  effect = {
    every_state = {
      if = {
        limit = { 
          state_population < 10000 
          NOT = { has_modifier = start_boost_country_modifier4 }
        }
        add_modifier = {
          name = start_boost_country_modifier4
          duration = 0
        }
      }
      if = {
        limit = {
          state_population > 30000
          has_modifier = start_boost_country_modifier4
        }
        remove_modifier = start_boost_country_modifier4
      }
    }
  }
}
