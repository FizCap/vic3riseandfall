on_yearly_pulse_country = {
  on_actions = {
    ai_improvement
  }
}

on_monthly_pulse_country = {
    on_actions = {
        ai_improvement_subsidies
    }
}

# Economic Boost Game Rule Effects

ai_improvement = {
    trigger = {
        has_game_rule = ai_improvement_enabled
        is_ai = yes
    }
    effect = {
        rf_ai_improvement_effect_dynamic = yes
    }
}

ai_improvement_subsidies = {
    trigger = {
        has_game_rule = ai_improvement_enabled
        is_ai = yes
    }
    effect = {
        rf_ai_building_subsidized = yes
    }
}
