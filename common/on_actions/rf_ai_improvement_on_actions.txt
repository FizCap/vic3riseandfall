on_yearly_pulse_country = {
  on_actions = {
    ai_improvement_lite
  }
}

on_monthly_pulse_country = {
    on_actions = {
        ai_improvement_on
        ai_improvement_subsidies
    }
}

# Lite mode keeps the expensive AI economy pass on the yearly pulse.

ai_improvement_lite = {
    trigger = {
        has_game_rule = ai_improvement_lite
        is_ai = yes
    }
    effect = {
        rf_ai_improvement_effect_dynamic = yes
    }
}

# Full mode runs the same AI economy pass every month.

ai_improvement_on = {
    trigger = {
        has_game_rule = ai_improvement_on
        is_ai = yes
    }
    effect = {
        rf_ai_improvement_effect_dynamic = yes
    }
}

ai_improvement_subsidies = {
    trigger = {
        rf_ai_improvement_active = yes
        is_ai = yes
    }
    effect = {
        rf_ai_building_subsidized = yes
    }
}
