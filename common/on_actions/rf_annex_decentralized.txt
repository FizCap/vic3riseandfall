on_monthly_pulse = {
    on_actions = {
        rf_annex_decentralized_monthly
    }
}

rf_annex_decentralized_monthly = {
    trigger = {
        always = yes
    }
    effect = {
        if = {
            limit = {
                year <= 1447
                has_game_rule = annexation_unlimited
            }
            rf_annex_decentralized_unlimited_effect = yes
        }
        else_if = {
            limit = {
                year <= 1447
                has_game_rule = annexation_limited
            }
            rf_annex_decentralized_limited_effect = yes
        }
    }
}
