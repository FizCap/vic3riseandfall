on_monthly_pulse_country = {
    on_actions = {
        rf_ai_auto_build_monthly_gov
        rf_ai_auto_build_monthly_rail
        rf_ai_auto_build_monthly_port
        rf_ai_auto_build_monthly_construction
        rf_auto_build_monthly_gov
        rf_auto_build_monthly_rail
        rf_auto_build_monthly_port
        rf_auto_build_monthly_power
        rf_auto_build_monthly_university
        rf_auto_build_monthly_construction
    }
}

rf_ai_auto_build_monthly_gov = {
    trigger = {
        rf_ai_improvement_active = yes
        is_ai = yes
    }
    effect = {
        if = {
            limit = {
                NOT = {
                    any_scope_state = {
                        b:building_government_administration ?= {
                            is_under_construction = yes
                        }
                    }
                }
            }
            rf_auto_build_government_administration = yes
        }
    }
}

rf_ai_auto_build_monthly_rail = {
    trigger = {
        rf_ai_improvement_active = yes
        is_ai = yes
    }
    effect = {
        if = {
            limit = {
                gold_reserves > rf_autobuild_reserve_limit
                bureaucracy > 0
                NOT = {
                    any_scope_state = {
                        b:building_railway ?= {
                            is_under_construction = yes
                        }
                    }
                }
            }
            rf_manual_build_railway = yes
        }
    }
}

rf_ai_auto_build_monthly_construction = {
    trigger = {
        rf_ai_improvement_active = yes
        is_ai = yes
    }
    effect = {
        if = {
            limit = {
                rf_can_expand_construction_sector = yes
                gold_reserves > rf_autobuild_reserve_limit
                rf_weekly_net_value > 0
                rf_positive_surplus_months_value >= 6
                rf_reserve_income_ratio <= 12
                NOT = {
                    any_scope_state = {
                        b:building_construction_sector ?= {
                            is_under_construction = yes
                        }
                    }
                }
            }
            rf_manual_build_construction_sector = yes
        }
    }
}

rf_auto_build_monthly_gov = {
    trigger = {
        is_player = yes
        has_variable = rf_auto_build_gov_cooldown
    }
    effect = {
        rf_player_auto_build_government_administration_scaled = yes
    }
}

rf_auto_build_monthly_rail = {
    trigger = {
        is_player = yes
        has_variable = rf_auto_build_rail_cooldown
    }
    effect = {
        # Railway candidates already avoid states with active railway construction.
        # Do not let one active railway block infrastructure fixes elsewhere.
        rf_player_auto_build_railway_scaled = yes
    }
}
