on_monthly_pulse_country = {
    on_actions = {
        rf_dynamic_country_ai_monthly
    }
}

on_monthly_pulse_state = {
    on_actions = {
        rf_recent_state_occupation_monthly
    }
}

rf_dynamic_country_ai_monthly = {
    trigger = {
        has_game_rule = ai_improvement_enabled
        OR = {
            is_country_type = recognized
            is_country_type = unrecognized
        }
        is_player = no
        is_subject = no
        NOT = {
            any_country = {
                OR = {
                    play_participant_has_war_goal_of_type_against = { type = annex_country target = ROOT }
                    war_participant_has_war_goal_of_type_against = { type = annex_country target = ROOT }
                }
            }
        }
        num_states > 1
        # Performance: only run for countries that have at least one state showing
        # significant turmoil or very low market access (candidates for dynamic country creation)
        any_scope_state = {
            owner = root
            NOT = { is_treaty_port = yes }
            OR = {
                turmoil > 0.30
                market_access < 0.15
            }
        }
    }
    effect = {
        rf_dynamic_country_ai = yes
    }
}

rf_recent_state_occupation_monthly = {
    trigger = {
        NOT = { controller = owner }
    }
    effect = {
        rf_state_recent_occupation_effect = yes
    }
}
