on_monthly_pulse_country = {
    on_actions = {
        rf_dynamic_country_ai_monthly
    }
}

on_monthly_pulse_state = {
    on_actions = {
        rf_recent_state_occupation_monthly
    }
}

rf_dynamic_country_ai_monthly = {
    trigger = {
        # Keep AI dynamic country creation controlled separately from the general AI economy helper.
        OR = {
            has_game_rule = dynamic_country_ai_creation_roleplay
            has_game_rule = dynamic_country_ai_creation_gameplay
        }
        OR = {
            is_country_type = recognized
            is_country_type = unrecognized
        }
        NOT = { is_subject_type = subject_type_chartered_company }
        # AI subjects should not recursively create their own dynamic subjects; players can still use the manual flow.
        is_subject = no
        is_player = no
        NOT = {
            any_country = {
                OR = {
                    play_participant_has_war_goal_of_type_against = { type = annex_country target = ROOT }
                    war_participant_has_war_goal_of_type_against = { type = annex_country target = ROOT }
                }
            }
        }
        num_states > 1
        # Performance: only run for countries that already have at least one
        # state matching the same candidate filter used by rf_dynamic_country_ai.
        any_scope_state = {
            owner = ROOT
            controller = owner
            is_capital = no
            is_under_colonization = no
            NOT = { is_treaty_port = yes }
            NOT = { has_variable = riseandfall_recent_state_owner_change }
            NOT = { has_variable = riseandfall_recent_state_occupation }
            NOT = { is_target_of_wargoal = ROOT }
            OR = {
                AND = {
                    # Roleplay mode lets AI split off non-primary-culture land without unrest pressure.
                    ROOT = { has_game_rule = dynamic_country_ai_creation_roleplay }
                    NOT = { is_homeland_of_country_cultures = ROOT }
                }
                AND = {
                    # Gameplay mode keeps the previous turmoil and market-access safeguards.
                    ROOT = { has_game_rule = dynamic_country_ai_creation_gameplay }
                    OR = {
                        NOT = { is_homeland_of_country_cultures = ROOT }
                        AND = {
                            market_access < 0.01
                        }
                    }
                    OR = {
                        AND = {
                            ROOT = { is_country_type = recognized }
                            OR = {
                                turmoil > 0.50
                                market_access < 0.15
                            }
                        }
                        AND = {
                            ROOT = { is_country_type = unrecognized }
                            OR = {
                                turmoil > 0.50
                                market_access < 0.15
                            }
                        }
                    }
                }
            }
        }
    }
    effect = {
        rf_dynamic_country_ai = yes
    }
}

rf_recent_state_occupation_monthly = {
    trigger = {
        NOT = { controller = owner }
    }
    effect = {
        rf_state_recent_occupation_effect = yes
    }
}
