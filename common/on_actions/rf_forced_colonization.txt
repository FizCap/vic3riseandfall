on_yearly_pulse_country = {
    on_actions = {
        rf_forced_colonization_yearly
    }
}

rf_forced_colonization_yearly = {
    trigger = {
        has_game_rule = rf_forced_colonization_enabled
        NOT = { is_country_type = decentralized }
        has_technology_researched = civilizing_mission
        has_law = law_type:law_no_colonial_affairs
        NOT = { has_variable = rf_forced_colonization_declined }
        NOT = { has_variable = rf_forced_colonization_pending }
        OR = {
            # Prefer countries with direct claims on neighboring decentralized frontiers.
            any_scope_state = {
                any_neighbouring_state = {
                    NOT = { is_under_colonization = yes }
                    has_claim_by = ROOT
                    owner = {
                        is_country_type = decentralized
                    }
                }
            }
            # Otherwise, only prompt countries when no neighboring colonizer is already contesting the frontier.
            any_scope_state = {
                any_neighbouring_state = {
                    NOT = { is_under_colonization = yes }
                    owner = {
                        is_country_type = decentralized
                        NOT = {
                            any_scope_state = {
                                has_claim_by = ROOT
                            }
                        }
                    }
                    NOT = {
                        any_neighbouring_state = {
                            owner = {
                                NOT = { this = ROOT }
                                NOT = { is_country_type = decentralized }
                                OR = {
                                    has_law_or_variant = law_type:law_frontier_colonization
                                    has_law_or_variant = law_type:law_colonial_resettlement
                                    has_law_or_variant = law_type:law_colonial_exploitation
                                    is_enacting_law = law_type:law_frontier_colonization
                                    is_enacting_law = law_type:law_colonial_resettlement
                                    is_enacting_law = law_type:law_colonial_exploitation
                                }
                            }
                        }
                    }
                }
            }
        }
    }
    effect = {
        rf_forced_colonization_apply_pressure = yes
    }
}
