on_yearly_pulse_character = {
  on_actions = {
    rf_random_character_death
  }
}

on_colony_created = {
    on_actions = {
        colony_grant_claim
    }
}

on_state_owner_change = {
    on_actions = {
        rf_state_owner_change
    }
}

# --- Event Definitions ---

rf_random_character_death = {
  trigger = {
    is_character_alive = yes
    has_game_rule = death_enabled
  }
  effect = {
    rf_random_character_death_effect = yes
  }
}

colony_grant_claim = {
  trigger = {
    has_game_rule = colony_grant_claim_enabled
  }
  effect = {
    rf_grant_colony_claim_effect = yes
  }
}

rf_state_owner_change = {
  effect = {
    rf_state_owner_change_effect = yes
  }
}
