et_scripted_button_auto_build_admin_toggle = {
    name = "et_scripted_button_auto_build_admin_toggle"
    desc = "et_scripted_button_auto_build_admin_toggle_desc"
    
    effect = {
        if = {
            limit = { has_variable = et_auto_build_admin_enabled }
            remove_variable = et_auto_build_admin_enabled
        }
        else = {
            set_variable = et_auto_build_admin_enabled
        }
    }
    
    visible = {
        always = yes
    }
    
    possible = {
        always = yes
    }
}
