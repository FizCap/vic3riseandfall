et_scripted_button_build_admin_center = {
    name = "et_scripted_button_build_admin_center"
    desc = "et_scripted_button_build_admin_center_desc"
    
    effect = {
        et_manual_build_government_administration = yes
    }
    
    visible = {
        always = yes
    }
    
    possible = {
            can_construct_building = building_government_administration
        }

    ai_chance = {
        value = 0
        if = {
            limit = {
                has_game_rule = ai_improvement_enabled
                gold_reserves > 10000
                bureaucracy < 0
                NOT = {
                    any_scope_state = {
                        b:building_government_administration ?= {
                            is_under_construction = yes
                        }
                    }
                }
            }
            add = 100
        }
    }
}

    name = "et_scripted_button_build_construction_sector"
    desc = "et_scripted_button_build_construction_sector_desc"

    effect = {
        et_manual_build_construction_sector = yes
    }

    visible = { always = yes }

    possible = {
        can_construct_building = building_construction_sector
    }

    ai_chance = {
        value = 0
        if = {
            limit = {
                has_game_rule = ai_improvement_enabled
                NOT = {
                    any_scope_state = {
                        b:building_construction_sector ?= {
                            is_under_construction = yes
                        }
                    }
                }
                AND = {
                    has_wasted_construction = no
                    investment_pool_net_income > 1
                }
            }
            add = 100
        }
    }
}
