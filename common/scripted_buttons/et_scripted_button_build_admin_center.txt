et_scripted_button_build_admin_center = {
    name = "et_scripted_button_build_admin_center"
    desc = "et_scripted_button_build_admin_center_desc"

    cooldown = { days = 60 }
    
    effect = {
        et_manual_build_government_administration = yes
    }
    
    visible = {
        always = yes
    }
    
    possible = {
            can_construct_building = building_government_administration
        }

    ai_chance = {
        value = 0
        if = {
            limit = { has_game_rule = ai_improvement_disabled }
            subtract = -1000
        }
        if = {
            limit = {
                has_game_rule = ai_improvement_enabled
                gold_reserves > 10000
                bureaucracy < 0
                NOT = {
                    any_scope_state = {
                        b:building_government_administration ?= {
                            is_under_construction = yes
                        }
                    }
                }
            }
            add = 100
        }
    }
}

et_scripted_button_auto_build_admin_toggle = {
    name = "et_scripted_button_auto_build_admin_toggle"
    desc = "et_scripted_button_auto_build_admin_toggle_desc"
    
    effect = {
        if = {
            limit = { has_variable = et_auto_build_gov_cooldown }
            remove_variable = et_auto_build_gov_cooldown
        }
        else = {
            set_variable = { which = et_auto_build_gov_cooldown value = 0 }
        }
    }
    
    visible = {
        always = yes
    }
    
    possible = {
        always = yes
    }
}
