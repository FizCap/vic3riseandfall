rf_scripted_button_build_admin_center = {
    name = "rf_scripted_button_build_admin_center"
    desc = "rf_scripted_button_build_admin_center_desc"

    effect = {
        rf_manual_build_government_administration = yes
    }

    visible = {
        is_player = yes
    }

    possible = {
        OR = {
            AND = {
                bureaucracy < 0
                rf_can_manually_queue_admin_bureaucracy_construction = yes
                any_scope_state = {
                    rf_admin_center_bureaucracy_state_candidate = yes
                }
            }
            AND = {
                rf_can_manually_queue_admin_tax_capacity_construction = yes
                any_scope_state = {
                    rf_admin_center_tax_capacity_state_candidate = yes
                }
            }
        }
    }

    ai_chance = { value = 0 }
}
