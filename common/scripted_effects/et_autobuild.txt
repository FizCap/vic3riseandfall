et_ai_auto_build_government_administration = {
    #auto build government administration by running manual effect every 6 months
    if = {
        limit = { NOT = { has_variable = et_auto_build_gov_cooldown } }
        et_manual_build_government_administration = yes
        set_variable = { which = et_auto_build_gov_cooldown value = 6 }
    }
    change_variable = { which = et_auto_build_gov_cooldown value = -1 }
    if = {
        limit = { et_auto_build_gov_cooldown <= 0 }
        remove_variable = et_auto_build_gov_cooldown
    }
}

et_manual_build_government_administration = {
    #manual build button for the government administration
    every_scope_state = {
        limit = {
            any_scope_building = {
                is_building_type = building_government_administration
                occupancy > 0.25
            }
            can_construct_building = building_government_administration
        }
        start_building_construction = building_government_administration
    }
}

et_manual_build_railway = {
    #manual build for railway, doesnt work
    every_scope_state = {
        limit = {
            infrastructure_delta < 0
            can_construct_building = building_port
        }
        start_building_construction = building_railway
    }
}

et_manual_build_port = {
    #manual build port in coastal state with low infrastructure
    every_scope_state = {
        limit = {
            infrastructure_delta < 0
            can_construct_building = building_port
        }
        start_building_construction = building_port
    }
}

et_auto_build_all = {
    #auto build all infrastructure when needed
    et_ai_auto_build_government_administration = yes
    et_ai_auto_build_railway = yes
    et_ai_auto_build_port = yes
}
