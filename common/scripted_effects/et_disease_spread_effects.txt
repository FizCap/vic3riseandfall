# Scope: state

            random_neighbouring_state = {
                limit = {

                        NOT = { has_harvest_condition = et_epidemic_outbreak }
                    }
                }
                random_list = {
                    99 = { state_region = { start_harvest_condition = et_epidemic_outbreak } }
                    1 = { }
                }
            }
        }
    }
