rf_manual_build_government_administration = {
    # Fix bureaucracy first; only rich, stable economies invest in broader tax-capacity coverage.
    if = {
        limit = {
            bureaucracy < 0
            rf_can_queue_admin_bureaucracy_construction = yes
        }
        rf_build_best_admin_center_bureaucracy_state = yes
    }

    if = {
        limit = {
            rf_can_queue_admin_tax_capacity_construction = yes
        }
        rf_build_best_admin_center_tax_capacity_state = yes
    }
}

rf_build_best_admin_center_tax_capacity_state = {
    # Tax-capacity shortages are the best target when the economy can afford long-term administration growth.
    ordered_state = {
        limit = {
            rf_admin_center_tax_capacity_state_candidate = yes
        }
        order_by = rf_tax_capacity_shortfall_value
        position = 0
        check_range_bounds = no
        start_building_construction = building_government_administration
    }
}

rf_manual_build_university = {
    # Queue a limited university batch so the player button improves coverage without overwhelming the budget.
    if = {
        limit = {
            rf_can_queue_university_construction = yes
        }
        rf_build_best_university_state = yes
    }

    if = {
        limit = {
            rf_can_queue_university_construction = yes
            rf_positive_surplus_months_value >= 6
            construction_queue_num_queued_government_levels < 9
            construction_queue_government_duration < 39
        }
        rf_build_best_university_state = yes
    }

    if = {
        limit = {
            rf_can_queue_university_construction = yes
            rf_positive_surplus_months_value >= 9
            rf_reserve_income_ratio <= 12
            construction_queue_num_queued_government_levels < 6
            construction_queue_government_duration < 26
        }
        rf_build_best_university_state = yes
    }

    if = {
        limit = {
            rf_can_queue_university_construction = yes
            rf_positive_surplus_months_value >= 12
            rf_reserve_income_ratio <= 6
            construction_queue_num_queued_government_levels < 3
            construction_queue_government_duration < 13
        }
        rf_build_best_university_state = yes
    }
}

rf_manual_build_construction_sector = {
    # Manual / Auto build for construction sector (prioritised: coastal first, then inland)

    # Prefer a coastal state with positive infrastructure so each new sector is placed where it can be staffed and used.
    if = {
        limit = {
            any_scope_state = {
                turmoil < 0.50
                is_coastal = yes
                infrastructure_delta > 0
                can_construct_building = building_construction_sector
                owner = ROOT
            }
        }
        random_state = {
            limit = {
                turmoil < 0.50
                is_coastal = yes
                infrastructure_delta > 0
                can_construct_building = building_construction_sector
                owner = ROOT
            }
            start_building_construction = building_construction_sector
        }
    }
    else = {
        # Fall back script if none are met
        if = {
            limit = {
                any_scope_state = {
                    turmoil < 0.50
                    can_construct_building = building_construction_sector
                    owner = ROOT
                }
            }
            random_state = {
                limit = {
                    turmoil < 0.50
                    can_construct_building = building_construction_sector
                    owner = ROOT
                }
                start_building_construction = building_construction_sector
            }
        }
    }
}
