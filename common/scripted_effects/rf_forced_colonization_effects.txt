# Eligibility is handled by the on_action trigger; this effect only applies the country response.
rf_forced_colonization_apply_pressure = {
    if = {
        limit = { is_ai = yes }
        activate_law = law_type:law_frontier_colonization
    }
    else = {
        set_variable = {
            name = rf_forced_colonization_pending
            days = 7
        }
        trigger_event = {
            id = rf_forced_colonization.1
            popup = yes
        }
    }
}
