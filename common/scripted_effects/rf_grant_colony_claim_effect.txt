rf_grant_colony_claim_effect = {
    if = {
        limit = {
            exists = owner
            is_under_colonization = yes
        }
        owner = { save_scope_as = colonizer }
        state_region = {
            if = {
                limit = {
                    NOT = { has_claim_by = scope:colonizer }
                }
                add_claim = scope:colonizer
            }
        }
    }
}
