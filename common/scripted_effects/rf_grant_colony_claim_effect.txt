rf_grant_colony_claim_effect = {
    if = {
        limit = {
            exists = owner
            is_under_colonization = yes
        }
        owner = { save_scope_as = colonizer }
        state_region = { save_scope_as = target_region }
        if = {
            limit = {
                scope:colonizer = {
                    NOT = { has_claim = scope:target_region }
                }
            }
            scope:target_region = {
                add_claim = scope:colonizer
            }
        }
    }
}
