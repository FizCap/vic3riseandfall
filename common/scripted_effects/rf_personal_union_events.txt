    c:AUS = {
        create_diplomatic_pact = {
            country = c:X59
            type = personal_union
        }
        trigger_event = { id = rf_austria_burgundy.1 }
    }
    c:X59 = {
        trigger_event = { id = rf_austria_burgundy.1 }

    }
}
