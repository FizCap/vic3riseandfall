# Scripted effect for applying stronger rebellion modifier during civil war/revolution
rf_stronger_rebellions_civil_script = {
    # ROOT is the subject country that spawned the rebellion.
    # scope:target is the rebel country from on_revolution_start.
    ROOT = {
        save_scope_as = rf_subject_country

        if = {
            limit = {
                is_subject = yes
            }

            top_overlord = {
                save_scope_as = rf_subject_overlord
            }

            scope:target ?= {
                if = {
                    limit = {
                        any_primary_culture = {
                            scope:rf_subject_country = {
                                country_has_primary_culture = PREV
                            }
                            scope:rf_subject_overlord = {
                                NOT = { country_has_primary_culture = PREV }
                            }
                        }
                    }
                    add_modifier = {
                        name = rf_stronger_rebellions
                    }
                }
            }
        }
    }
}
