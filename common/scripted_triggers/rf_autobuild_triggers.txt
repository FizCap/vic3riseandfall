rf_can_expand_construction_sector = {
    # Only add construction capacity after the government queue has real backlog.
    # This prevents rich AIs from paying for idle construction sectors when private building is the only active queue.
    has_wasted_construction = no
    construction_queue_num_queued_government_levels >= 3
    construction_queue_government_duration >= 12
}

rf_player_autobuild_slot_2 = {
    # Second player auto-build slot opens for economies with current weekly surplus and a manageable queue.
    rf_weekly_net_value > 0
    construction_queue_government_duration < 52
}

rf_player_autobuild_slot_3 = {
    # Stronger economies may keep queuing past the base one-year government queue target.
    rf_weekly_net_share_of_income >= 0.05
    rf_autobuild_effective_reserve_surplus > 0
    construction_queue_government_duration < 78
}

rf_player_autobuild_slot_4 = {
    # Very strong economies can tolerate a longer government queue while reserves remain healthy.
    rf_weekly_net_share_of_income >= 0.10
    rf_reserve_income_ratio <= 12
    construction_queue_government_duration < 104
}

rf_player_autobuild_slot_5 = {

    rf_reserve_income_ratio <= 6
    construction_queue_government_duration < 156
}
