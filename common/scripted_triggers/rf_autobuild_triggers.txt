rf_can_expand_construction_sector = {
    # Only add construction capacity after the government queue has real backlog.
    # This prevents rich AIs from paying for idle construction sectors when private building is the only active queue.
    has_wasted_construction = no
    construction_queue_num_queued_government_levels >= 3
    construction_queue_government_duration >= 12
}

rf_can_queue_admin_bureaucracy_construction = {
    # Bureaucracy fixes are allowed earlier than tax-capacity investment, but still avoid piling onto an overloaded queue.
    gold_reserves > 10000
    construction_queue_num_queued_government_levels < 12
    construction_queue_government_duration < 52
}

rf_can_queue_admin_tax_capacity_construction = {
    # Tax-capacity campaigns are expensive in poor high-population countries, so reserve them for strongly booming economies.
    bureaucracy > 100
    gold_reserves > rf_autobuild_reserve_limit
    rf_weekly_net_value > 0
    rf_positive_surplus_months_value >= 9
    rf_reserve_income_ratio <= 6
    construction_queue_num_queued_government_levels < 6
    construction_queue_government_duration < 26
}

rf_admin_center_bureaucracy_state_candidate = {
    owner = ROOT
    turmoil < 0.35
    can_construct_building = building_government_administration
    NOT = {
        b:building_government_administration ?= {
            is_under_construction = yes
        }
    }
}

rf_admin_center_tax_capacity_state_candidate = {
    rf_admin_center_bureaucracy_state_candidate = yes
    is_incorporated = yes
    tax_capacity < tax_capacity_usage
}

rf_can_queue_university_construction = {
    # Universities are government-funded, so require a stable surplus and a queue that is not already overloaded.
    gold_reserves > rf_autobuild_reserve_limit
    rf_weekly_net_value > 0
    rf_positive_surplus_months_value >= 3
    construction_queue_num_queued_government_levels < 12
    construction_queue_government_duration < 52
}

rf_university_state_candidate = {
    owner = ROOT
    turmoil < 0.25
    can_construct_building = building_university
    NOT = {
        b:building_university ?= {
            level_after_queued_constructions >= 1
        }
    }
}

rf_can_queue_power_plant_construction = {
    # Power plants are useful everywhere eventually, but they still add government construction and input costs.
    has_technology_researched = electrical_generation
    gold_reserves > rf_autobuild_reserve_limit
    rf_weekly_net_value > 0
    rf_positive_surplus_months_value >= 3
    construction_queue_num_queued_government_levels < 12
    construction_queue_government_duration < 52
}

rf_power_plant_missing_state_candidate = {
    owner = ROOT
    turmoil < 0.35
    can_construct_building = building_power_plant
    NOT = {
        b:building_power_plant ?= {
            level_after_queued_constructions >= 1
        }
    }
}

rf_power_plant_expansion_state_candidate = {
    owner = ROOT
    turmoil < 0.35
    can_construct_building = building_power_plant
    b:building_power_plant ?= {
        level_after_queued_constructions >= 1
    }
    OR = {
        b:building_power_plant ?= {
            weekly_profit > 500
        }
        sg:electricity = {
            state_goods_pricier > 0.25
        }
    }
}

rf_ai_improvement_active = {
    # Treat both AI Improvement modes as active for shared AI button behavior.
    OR = {
        has_game_rule = ai_improvement_lite
        has_game_rule = ai_improvement_on
    }
}
